def []=(key, value)
  return if @sent.has_key?(key) && !(ALLOWED_DUPLICATES.include?(key))

  value = case value
    when Time then value.httpdate
    when NilClass then return
    else value.to_s
  end

  @sent[key] = value
  @out << HEADER_FORMAT % [key, value]
end
