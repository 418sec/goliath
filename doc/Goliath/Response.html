def each
  yield head
  yield headers_output

  if body.respond_to?(:each)
    body.each { |chunk| yield chunk }
  else
    yield body
  end
end

def headers_output
  headers[CONNECTION] = CLOSE if @send_close
  headers[SERVER] = Goliath::Request::SERVER
  headers[DATE] = Time.now.httpdate

  "#{headers.to_s}\r\n"
end
